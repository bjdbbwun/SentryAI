# .github/workflows/update-feeds.yml
# Automatically collect fraud data every 6 hours

name: Auto-Collect Fraud Feeds

on:
  schedule:
    - cron: '0 */6 * * *'  # Every 6 hours
  workflow_dispatch:        # Manual trigger

jobs:
  collect-threats:
    runs-on: ubuntu-latest
    
    steps:
      - name: Checkout code
        uses: actions/checkout@v4
      
      - name: Setup Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'
      
      - name: Install dependencies
        run: |
          pip install -r python/requirements.txt
      
      - name: Run fraud collector
        run: |
          python python/collector.py
      
      - name: Run pattern learner
        run: |
          python python/auto_learn.py
      
      - name: Commit and push changes
        run: |
          git config user.name "github-actions[bot]"
          git config user.email "github-actions[bot]@users.noreply.github.com"
          git add data/
          git commit -m "🤖 Auto-update: fresh fraud data collected" || exit 0
          git push
